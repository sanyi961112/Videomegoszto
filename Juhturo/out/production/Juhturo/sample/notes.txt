CREATE TABLE FELHASZNALOK (
   fh_nev VARCHAR2(30) NOT NULL,
   jelszo VARCHAR2(64) NOT NULL,
   teljes_nev VARCHAR2(64) NOT NULL,
	email VARCHAR2(30) NOT NULL,
	PRIMARY KEY(fh_nev)
 );

 CREATE TABLE video (
	video_id VARCHAR(10) NOT NULL,
	cim VARCHAR2(128) NOT NULL,
	leiras VARCHAR2(500),
	feltoltes_datuma DATE NOT NULL,
	feltolto VARCHAR2(30) NOT NULL,
	forras_fajl VARCHAR(80) NOT NULL,
	kategoria VARCHAR(100) NOT NULL,
	megtekintes NUMBER,
	PRIMARY KEY(video_id),
	FOREIGN KEY(feltolto) REFERENCES felhasznalok(fh_nev)
 );

 CREATE TABLE kommentek (
	video_id VARCHAR(10) NOT NULL,
	fh_nev VARCHAR2(30) NOT NULL,
	idopont TIMESTAMP(2),
	komment VARCHAR2(200) NOT NULL,

	CONSTRAINT komment_prim
	PRIMARY KEY(video_id, fh_nev, idopont),

	CONSTRAINT komment_foreign1
	FOREIGN KEY(video_id) REFERENCES video(video_id),

	CONSTRAINT komment_foreign2
	FOREIGN KEY(fh_nev) REFERENCES felhasznalok(fh_nev)
 );

 CREATE TABLE kategoriak (
	kategoria_nev VARCHAR(30),
	PRIMARY KEY(kategoria_nev)
	);

 CREATE TABLE cimkek (
	video_id VARCHAR(10),
	cimke_nev VARCHAR(30),
	PRIMARY KEY(video_id, cimke_nev),
	FOREIGN KEY(video_id) REFERENCES video(video_id)
 );

 CREATE TABLE kedvelesek(
	video_id VARCHAR(10),
	fh_nev VARCHAR2(30),

	CONSTRAINT kedvelesek_prim
	PRIMARY KEY (video_id, fh_nev),

	CONSTRAINT kedveles_foreign1
	FOREIGN KEY(video_id) REFERENCES video(video_id),

	CONSTRAINT kedvelesek_foreign2
	FOREIGN KEY(fh_nev) REFERENCES felhasznalok(fh_nev)
 );

//end of tables
